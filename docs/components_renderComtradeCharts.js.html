/**
 * @module renderComtradeCharts
 * @description
 * Main chart rendering orchestrator for COMTRADE application.
 * Handles the complete workflow of rendering analog and digital charts
 * from parsed COMTRADE data, with support for drag-and-drop reordering
 * and vertical line overlays.
 *
 * @requires chartComponent - for buildChartOptions()
 * @requires createDragBar - for drag bar DOM elements
 * @requires setupChartDragAndDrop - for drag/drop functionality
 * @requires renderAnalogCharts - for analog chart rendering
 * @requires renderDigitalCharts - for digital chart rendering
 * @requires autoGroupChannels - for smart analog channel grouping
 * @requires createState - for reactive state management
 *
 * @example
 * import { renderComtradeCharts } from './components/renderComtradeCharts.js';
 *
 * const charts = [];
 * const verticalLinesX = []; // or createState([])
 *
 * renderComtradeCharts(
 *   cfg,                      // COMTRADE config object
 *   data,                     // Parsed COMTRADE data
 *   chartsContainer,          // DOM element for charts
 *   charts,                   // Array to store chart instances
 *   verticalLinesX,           // State for vertical line positions
 *   createState,              // State creation utility
 *   calculateDeltas,          // Function to calculate time/value deltas
 *   'seconds',                // TIME_UNIT
 *   channelState              // Channel metadata state
 * );
 */

// src/components/renderComtradeCharts.js
import { createChartOptions } from "./chartComponent.js";
import { createDragBar } from "./createDragBar.js";
import { setupChartDragAndDrop } from "./setupChartDragAndDrop.js";
import { renderAnalogCharts } from "./renderAnalogCharts.js";
import { renderDigitalCharts } from "./renderDigitalCharts.js";
import { destroyCharts } from "../utils/chartUtils.js";
import { autoGroupChannels } from "../utils/autoGroupChannels.js";
import { createDigitalFillPlugin } from "../plugins/digitalFillPlugin.js";
import { unwrap, createState } from "./createState.js";

/**
 * Render COMTRADE charts (analog and digital) in the container.
 *
 * Orchestrates the complete chart rendering workflow:
 * 1. Clears previous charts and DOM
 * 2. Sets up drag-and-drop functionality
 * 3. Renders analog charts with smart grouping
 * 4. Renders digital charts if present
 * 5. Calculates and displays time/value deltas for vertical lines
 *
 * @function renderComtradeCharts
 * @param {Object} cfg - COMTRADE configuration object
 * @param {Array} cfg.analogChannels - Array of analog channel definitions
 * @param {Array} cfg.digitalChannels - Array of digital channel definitions
 * @param {number} cfg.samplingRate - Sampling rate in samples/second
 * @param {Object} data - Parsed COMTRADE data
 * @param {Float64Array} data.time - Time array (uniform spacing)
 * @param {Array<Float64Array>} data.analogData - 2D array [channel][sample]
 * @param {Array<Float64Array>} data.digitalData - 2D array [channel][sample]
 * @param {HTMLElement} chartsContainer - DOM element to render charts into
 * @param {Array} charts - Array reference to store uPlot instances
 * @param {Array|State} verticalLinesX - Array of vertical line X positions (optional, can be createState)
 * @param {Function} createState - Reactive state creator function
 * @param {Function} calculateDeltas - Function to calculate deltas at vertical lines
 * @param {string} TIME_UNIT - Time unit label ('seconds', 'milliseconds', etc)
 * @param {Object} channelState - Reactive state for channel metadata (colors, labels, scales, units, inverts)
 * @returns {void}
 *
 * @example
 * renderComtradeCharts(
 *   cfg,
 *   data,
 *   document.getElementById('charts'),
 *   charts,
 *   verticalLinesX,
 *   createState,
 *   calculateDeltas,
 *   'seconds',
 *   channelState
 * );
 */
export function renderComtradeCharts(
  cfg,
  data,
  chartsContainer,
  charts,
  verticalLinesX,
  createState,
  calculateDeltas,
  TIME_UNIT,
  channelState
) {
  charts.length = 0; // <-- Clear the array at the start!
  chartsContainer.innerHTML = "";
  //destroyCharts(charts);
  setupChartDragAndDrop(chartsContainer);
  renderAnalogCharts(
    cfg,
    data,
    chartsContainer,
    charts,
    verticalLinesX,
    channelState,
    autoGroupChannels
  );
  if (
    cfg.digitalChannels &&
    cfg.digitalChannels.length > 0 &&
    data.digitalData &&
    data.digitalData.length > 0
  ) {
    renderDigitalCharts(
      cfg,
      data,
      chartsContainer,
      charts,
      verticalLinesX,
      channelState
    );
  }
  console.log("Charts rendered:", charts.length);
  if (charts.length > 0) {
    calculateDeltas(
      charts,
      verticalLinesX,
      charts[0].data[0],
      charts.map((c) => c.data[1])
    );
  }
  // Do NOT clear charts here!
}
